def run_descent(reg_param):
    alpha = 0.01
    tolerance = 1e-3
    theta_0 = np.zeros(X_poly.shape[1])
    Xval_fitted = main.fit(Xval_raw, poly, scaler)

    train_losses = []
    val_losses = []
    progress = None
    theta = None
    for i in range(1, len(X_poly) + 1):
        progress, theta = main.run_descent(alpha, tolerance, theta_0, X_poly[:i], y[:i], reg_param)
        train_losses.append(progress[-1])
        val_losses.append(main.linear_loss(theta, Xval_fitted, yval, 0))
    return (train_losses, val_losses, progress, theta)

reg_params = np.logspace(-2, 0, 10)
results = [run_descent(reg_param) for reg_param in reg_params]
